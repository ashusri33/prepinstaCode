import java.util.*;
public class Replace0with1
{
public static void main(String args[])
{
Scanner sc=new Scanner(System.in);
int n=sc.nextInt();
String str=Integer.toString(n);
int l=str.length();
int str1="";
for(int i=0;i<l;i++)
{
if(str.charAt(i)=='0')
  str1=str1+'1';
else 
  str1=str1+str.charAt(i);
}
System.out.println(str1);
}
}
